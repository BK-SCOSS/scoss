40	bool isPromisingPath(int curPos)
41	{
42	    int numCityLeft = n - curPos;
43	    return curTotalCost + numCityLeft * minCost <= m;
44	}

46	void tryPath(int curPos)
47	{
48	    if(curPos == n)
49	    {
50	        int prevCity = path[curPos - 1];
51	        curTotalCost += cost[prevCity][1];
52	        
53	        if(curTotalCost <= m)
54	            numSolution++;
55	        
56	        curTotalCost -= cost[prevCity][1];
57	        return;
58	    }
59	    
60	    int prevCity = path[curPos-1];
61	    for(int curCity = 1; curCity <= n; ++curCity)
62	    {
63	        if(!visited[curCity])
64	        {
65	            path[curPos] = curCity;
66	            curTotalCost += cost[prevCity][curCity];
67	            visited[curCity] = true;
68	            if(isPromisingPath(curPos))
69	                tryPath(curPos + 1);
70	            path[curPos] = 0;
71	            curTotalCost -= cost[prevCity][curCity];
72	            visited[curCity] = false;
73	        }
74	    }
75	}
